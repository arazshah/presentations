import json from datetime import datetime import statistics class SalesManager: def __init__(self, filename="sales_data.json"): self.filename = filename self.sales = self.load_data() def load_data(self): """بارگذاری داده‌ها از فایل""" try: with open(self.filename, 'r', encoding='utf-8') as file: return json.load(file) except FileNotFoundError: print("فایل داده یافت نشد. فایل جدید ایجاد می‌شود.") return [] except json.JSONDecodeError: print("خطا در خواندن فایل JSON") return [] def save_data(self): """ذخیره داده‌ها در فایل""" try: with open(self.filename, 'w', encoding='utf-8') as file: json.dump(self.sales, file, ensure_ascii=False, indent=2) print("داده‌ها ذخیره شدند") except Exception as e: print(f"خطا در ذخیره: {e}")
def add_sale(self, product_name, price, quantity=1): """اضافه کردن فروش جدید""" sale = { "id": len(self.sales) + 1, "product": product_name, "price": price, "quantity": quantity, "total": price * quantity, "date": datetime.now().strftime("%Y-%m-%d %H:%M:%S") } self.sales.append(sale) print(f"فروش ثبت شد: {product_name} - {sale['total']:,} تومان") def get_total_sales(self): """محاسبه کل فروش""" return sum(sale['total'] for sale in self.sales) def get_sales_by_product(self, product_name): """فروش یک محصول خاص""" product_sales = [sale for sale in self.sales if sale['product'].lower() == product_name.lower()] return product_sales def get_sales_stats(self): """آمار فروش""" if not self.sales: return "هیچ فروشی ثبت نشده" totals = [sale['total'] for sale in self.sales] stats = { "تعداد فروش": len(self.sales), "کل فروش": sum(totals), "میانگین فروش": statistics.mean(totals), "بیشترین فروش": max(totals), "کمترین فروش": min(totals) } return stats

class CSVAnalyzer: def __init__(self, filename): self.filename = filename self.data = [] self.headers = [] def load_csv(self): """بارگذاری فایل CSV""" try: with open(self.filename, 'r', encoding='utf-8') as file: lines = file.readlines() if not lines: print("فایل خالی است") return False # خواندن سرتیترها self.headers = lines[0].strip().split(',') # خواندن داده‌ها for line in lines[1:]: values = line.strip().split(',') if len(values) == len(self.headers): row = dict(zip(self.headers, values)) self.data.append(row) print(f"✅ {len(self.data)} رکورد بارگذاری شد") return True except FileNotFoundError: print(f"❌ فایل {self.filename} پیدا نشد") return False except Exception as e: print(f"❌ خطا در بارگذاری: {e}") return False
def analyze_column(self, column_name): """تحلیل یک ستون خاص""" if column_name not in self.headers: print(f"ستون {column_name} وجود ندارد") return None values = [row[column_name] for row in self.data if row[column_name]] # تشخیص نوع داده numeric_values = [] for value in values: try: numeric_values.append(float(value)) except ValueError: pass if numeric_values: # تحلیل عددی analysis = { "نوع": "عددی", "تعداد": len(numeric_values), "میانگین": sum(numeric_values) / len(numeric_values), "حداکثر": max(numeric_values), "حداقل": min(numeric_values), "مجموع": sum(numeric_values) } else: # تحلیل متنی unique_values = list(set(values)) analysis = { "نوع": "متنی", "تعداد": len(values), "منحصربه‌فرد": len(unique_values), "نمونه": unique_values[:5] } return analysis def generate_report(self): """تولید گزارش کامل""" if not self.data: print("هیچ داده‌ای بارگذاری نشده") return print("\n" + "="*50) print(f"📊 گزارش تحلیل فایل: {self.filename}") print("="*50) print(f"تعداد سطرها: {len(self.data)}") print(f"تعداد ستون‌ها: {len(self.headers)}") print(f"ستون‌ها: {', '.join(self.headers)}") print("\n📈 تحلیل ستون‌ها:") for header in self.headers: analysis = self.analyze_column(header) if analysis: print(f"\n🔹 {header}:") for key, value in analysis.items(): print(f" {key}: {value}") # استفاده analyzer = CSVAnalyzer("sales_data.csv") if analyzer.load_csv(): analyzer.generate_report()
