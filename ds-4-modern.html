Series - عملیات و متدها

ایجاد Series 

# از لیست
s1 = pd.Series([10, 20, 30, 40])
print(s1)

# با اندیس دلخواه
s2 = pd.Series([100, 200, 300], 
                index=['a', 'b', 'c'])
print(s2)

# از دیکشنری
data = {'علی': 25, 'مریم': 30, 'رضا': 28}
s3 = pd.Series(data)
print(s3)

# با نام و dtype مشخص
s4 = pd.Series([1, 2, 3], 
                name='شماره', dtype='int64')

عملیات اصلی روی Series 

# دسترسی به عناصر
print(s1[0])          # اولین عنصر
print(s1[1:3])       # برش
print(s2['a'])         # با اندیس برچسب

# عملیات ریاضی
print(s1 + 10)       # جمع با عدد
print(s1 * 2)        # ضرب در عدد
print(s1 + s1)        # جمع دو Series

# توابع آماری
print(s1.sum())       # مجموع
print(s1.mean())      # میانگین
print(s1.max())       # حداکثر
print(s1.min())       # حداقل
print(s1.describe())  # آمار توصیفی

🎯 ویژگی‌های کلیدی Series

اندیس (Index)

  • • هر Series یک اندیس دارد
  • • اندیس می‌تواند عددی یا متنی باشد
  • • اندیس‌ها تکراری نمی‌شوند

مقادیر (Values)

  • • مقادیر می‌توانند انواع مختلفی داشته باشند
  • • نوع داده با dtype مشخص می‌شود
  • • عملیات روی تمام مقادیر انجام می‌شود

DataFrame - ساختار اصلی داده

ایجاد DataFrame

data = {
    'نام': ['علی', 'مریم', 'رضا', 'سارا'],

💡 تفاوت loc و iloc

loc:

بر اساس برچسب‌ها (labels) کار می‌کند - شامل انتها

iloc:

بر اساس موقعیت‌های عددی کار می‌کند - شامل انتها نیست

فیلتر کردن سطرها 

# شرط ساده
young_people = df[df['سن'] < 30]
print(young_people)

# شرط متنی
tehranis = df[df['شهر'] == 'تهران']
print(tehranis)

# شرط ترکیبی (AND)
high_salary_tehran = df[
    (df['شهر'] == 'تهران') & 
    (df['حقوق'] > 5500)
]
print(high_salary_tehran)

# شرط ترکیبی (OR)
specific_cities = df[
    (df['شهر'] == 'تهران') | 
    (df['شهر'] == 'مشهد')
]
print(specific_cities)

متدهای پیشرفته فیلتر 

# با متد query
result = df.query('سن > 25 and حقوق < 6000')
print(result)

# با متد isin
selected_cities = df[df['شهر'].isin(['تهران', 'اصفهان'])]
print(selected_cities)

# با متد between
age_range = df[df['سن'].between(25, 30)]
print(age_range)

# با متد str.contains
names_with_a = df[df['نام'].str.contains('ا')]
print(names_with_a)

✅ عملگرهای منطقی

  • • & : AND
  • • | : OR
  • • ~ : NOT
  • • ^ : XOR

⚠️ نکته مهم

همیشه از پرانتز برای گروه‌بندی شروط استفاده کنید و از عملگرهای پایتون (and, or) در فیلتر کردن استفاده نکنید.

گروه‌بندی و تجمیع داده‌ها

GroupBy پایه 

# گروه‌بندی بر اساس یک ستون
grouped = df.groupby('شهر')

# مشاهده گروه‌ها
for name, group in grouped:
    print(f"شهر: {name}")
    print(group)
    print("---")

# گروه‌بندی بر اساس چند ستون
multi_grouped = df.groupby(['شهر', 'دسته_بندی_سن'])
print(multi_grouped.size())

توابع تجمیع (Aggregation) 

# تابع تجمیع ساده
city_stats = df.groupby('شهر')['حقوق'].mean()
print(city_stats)

# چند تابع تجمیع
multi_stats = df.groupby('شهر')['حقوق'].agg([
    'mean', 'median', 'std', 'count'
])
print(multi_stats)

# توابع تجمیع برای چند ستون
all_stats = df.groupby('شهر').agg({
    'حقوق': ['mean', 'max'],
    'سن': ['mean', 'min']
})
print(all_stats)

# تابع تجمیع سفارشی
def range_func(x):
    return x.max() - x.min()

custom_agg = df.groupby('شهر')['حقوق'].agg(range_func)
print(custom_agg)

متدهای مفید GroupBy 

# اندازه هر گروه
group_sizes = df.groupby('شهر').size()
print(group_sizes)

# اولین و آخرین هر گروه
first_items = df.groupby('شهر').first()
last_items = df.groupby('شهر').last()

# nth: عنصر n-ام هر گروه
nth_items = df.groupby('شهر').nth(1)  # عنصر دوم

# rank: رتبه‌بندی در هر گروه
df['رتبه_حقوق_در_شهر'] = df.groupby('شهر')['حقوق'].rank()

# cumsum: مجموع تجمعی در هر گروه
df['مجموع_حقوق_تجمعی'] = df.groupby('شهر')['حقوق'].cumsum()

# transform: اعمال تابع و حفظ شکل اصلی
df['میانگین_حقوق_شهر'] = df.groupby('شهر')['حقوق'].transform('mean')

📊 توابع تجمیع رایج

  • • sum(): مجموع
  • • mean(): میانگین
  • • median(): میانه
  • • std(): انحراف معیار
  • • count(): تعداد
  • • min(), max(): حداقل و حداکثر

💡 نکات کاربردی

  • • از as_index=False برای حفظ index
  • • reset_index() برای تبدیل index به ستون
  • • dropna() قبل از groupby برای داده‌های تمیز
  • • از agg() برای چند تابع همزمان

تمیز کردن داده‌ها

شناسایی مقادیر خالی (Missing Values) 

# بررسی مقادیر خالی
print(df.isnull().sum())

# درصد مقادیر خالی در هر ستون
missing_percentage = (df.isnull().sum() / len(df)) * 100
print(missing_percentage)

# نمایش سطرهای با مقادیر خالی
missing_rows = df[df.isnull().any(axis=1)]
print(missing_rows)

# بررسی مقادیر غیر خالی
print(df.notnull().sum())

# اطلاعات کلی درباره مقادیر خالی
print(df.info())

مدیریت مقادیر خالی 

# حذف سطرهای با مقادیر خالی
df_clean = df.dropna()

# حذف ستون‌های با مقادیر خالی
df_clean_cols = df.dropna(axis=1)

# حذف سطرهایی که همه مقادیرشان خالی است
df_clean_all = df.dropna(how='all')

# پر کردن مقادیر خالی با یک مقدار ثابت
df_filled = df.fillna(0)

# پر کردن با میانگین ستون
df_mean_fill = df.fillna(df.mean())

# پر کردن با مد (برای داده‌های دسته‌ای)
df_mode_fill = df.fillna(df.mode().iloc[0])

# پر کردن با مقدار قبلی (forward fill)
df_ffill = df.fillna(method='ffill')

# پر کردن با مقدار بعدی (backward fill)
df_bfill = df.fillna(method='bfill')

مدیریت مقادیر تکراری و نادرست 

# شناسایی سطرهای تکراری
duplicates = df.duplicated()

# نمایش سطرهای تکراری

df_no_duplicates = df.drop_duplicates()

# حذف تکراری‌ها بر اساس ستون‌های خاص
df_unique = df.drop_duplicates(subset=['نام', 'شهر'])

df_corrected = df.replace({

# حذف فضاهای خالی اضافی
df['نام'] = df['نام'].str.strip()

پارامترهای مهم

پارامترهای عمومی

  • • index=False: عدم ذخیره index
  • • encoding='utf-8': برای متن فارسی
  • • header=True: ذخیره نام ستون‌ها
  • • compression='gzip': فشرده‌سازی

پارامترهای CSV

  • • sep=',': جداکننده ستون‌ها
  • • decimal='.': جداکننده اعشار
  • • quotechar='"': کاراکتر نقل قول
  • • na_rep='NA': نمایش NaN

💡 نکات مهم

  • • برای فایل‌های بزرگ از فرمت‌های بهینه مانند Parquet یا Feather استفاده کنید
  • • همیشه مسیر ذخیره را بررسی کنید و از وجود پوشه مطمئن شوید
  • • برای داده‌های فارسی حتماً از encoding='utf-8' استفاده کنید
  • • پس از ذخیره، فایل را باز کرده و صحت آن را بررسی کنید

پروژه عملی - بخش اول

تحلیل داده‌های فروش یک شرکت

شرح پروژه

در این پروژه، ما داده‌های فروش یک شرکت را تحلیل می‌کنیم. داده‌ها شامل اطلاعات فروش محصولات در ماه‌های مختلف، قیمت‌ها، تعداد فروش، و اطلاعات مشتریان است. هدف اصلی، تحلیل عملکرد فروش و ارائه گزارش‌های مدیریتی است.

ساختار داده‌ها 

# ایجاد داده‌های نمونه
import pandas as pd
import numpy as np

# داده‌های فروش
sales_data = {
    'تاریخ': pd.date_range(start='2023-01-01', end='2023-12-31', freq='D'),
    'محصول': np.random.choice(['لپتاپ', 'موبایل', 'تبلت', 'مانیتور'], 365),
    'تعداد': np.random.randint(1, 50, 365),
    'قیمت_واحد': np.random.choice([50000, 20000, 15000, 8000], 365),
    'شهر': np.random.choice(['تهران', 'اصفهان', 'مشهد', 'شیراز'], 365),
    'فروشنده': np.random.choice(['علی', 'مریم', 'رضا', 'سارا'], 365)
}

df_sales = pd.DataFrame(sales_data)
df_sales['مبلغ_کل'] = df_sales['تعداد'] * df_sales['قیمت_واحد']

تحلیل فروش ماهانه 

# اضافه کردن ستون ماه و سال
df_sales['ماه'] = df_sales['تاریخ'].dt.month
df_sales['سال'] = df_sales['تاریخ'].dt.year
df_sales['نام_ماه'] = df_sales['تاریخ'].dt.strftime('%B')

# تحلیل فروش ماهانه
monthly_sales = df_sales.groupby(['سال', 'ماه', 'نام_ماه']).agg({
    'مبلغ_کل': 'sum',
    'تعداد': 'sum'
}).reset_index()

print("فروش ماهانه:")
print(monthly_sales)

# بهترین و بدترین ماه‌ها
best_month = monthly_sales.loc[monthly_sales['مبلغ_کل'].idxmax()]
worst_month = monthly_sales.loc[monthly_sales['مبلغ_کل'].idxmin()]

print(f"\nبهترین ماه فروش: {best_month['نام_ماه']} با مبلغ {best_month['مبلغ_کل']:,.0f}")
print(f"بدترین ماه فروش: {worst_month['نام_ماه']} با مبلغ {worst_month['مبلغ_کل']:,.0f}")

تحلیل عملکرد فروشندگان 

# تحلیل عملکرد فروشندگان
seller_performance = df_sales.groupby('فروشنده').agg({
    'مبلغ_کل': ['sum', 'mean', 'count'],
    'تعداد': 'sum'
}).round(2)

# تغییر نام ستون‌ها
seller_performance.columns = ['مجموع_فروش', 'میانگین_فروش', 'تعداد_تراکنش', 'تعداد_کالا']

# رتبه‌بندی فروشندگان
seller_performance['رتبه'] = seller_performance['مجموع_فروش'].rank(ascending=False)
seller_performance = seller_performance.sort_values('رتبه')

print("عملکرد فروشندگان:")
print(seller_performance)

# بهترین فروشنده
best_seller = seller_performance.index[0]
print(f"\nبهترین فروشنده: {best_seller}")

تحلیل محصولات پرفروش 

# تحلیل محصولات
product_analysis = df_sales.groupby('محصول').agg({
    'مبلغ_کل': 'sum',
    'تعداد': 'sum',
    'قیمت_واحد': 'mean'
}).round(2)

product_analysis.columns = ['مجموع_فروش', 'تعداد_فروش', 'میانگین_قیمت']
product_analysis = product_analysis.sort_values('مجموع_فروش', ascending=False)

# محاسبه سهم هر محصول
product_analysis['سهم_درصدی'] = (
    product_analysis['مجموع_فروش'] / 
    product_analysis['مجموع_فروش'].sum() * 100
).round(2)

print("تحلیل محصولات:")
print(product_analysis)

# محصول پرفروش
top_product = product_analysis.index[0]
print(f"\nمحصول پرفروش: {top_product}")
