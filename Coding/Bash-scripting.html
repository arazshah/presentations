                show_header
                systemctl list-units --type=service --state=running | head -20
                press_key

                press_key

                press_key
                ;;
            5)
                read -p "Service name: " svc
                systemctl status "$svc"
                press_key
                ;;
            0) break ;;
        esac
    done
}

# Main loop
main() {
    while true; do
        show_header
        show_menu
        
        read -p "Enter choice [0-9]: " choice
        
        case $choice in
            1) system_info ;;
            2) disk_usage ;;
            3) memory_usage ;;
            4) network_info ;;
            5) process_menu ;;
            6) echo "User management..." ; press_key ;;
            7) service_menu ;;
            8) echo "Log viewer..." ; press_key ;;
            9) echo "Backup menu..." ; press_key ;;
            0)
                echo -e "${GREEN}Goodbye!${NC}"
                exit 0
                ;;
            *)
                echo -e "${RED}Invalid option${NC}"
                sleep 1
                ;;
        esac
    done
}
