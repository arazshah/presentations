class Trainer:
    """Ú©Ù„Ø§Ø³ Ø¢Ù…ÙˆØ²Ø´"""
    
    def __init__(self, model, device):
        self.model = model.to(device)
        self.device = device
        
        # Loss
        self.dice_loss = smp.losses.DiceLoss(mode='multiclass')
        self.ce_loss = nn.CrossEntropyLoss()
        
        # Optimizer
        self.optimizer = torch.optim.AdamW(
            model.parameters(), 
            lr=CONFIG['learning_rate'],
            weight_decay=0.01
        )
        
        # Scheduler
        self.scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(
            self.optimizer, 
            T_max=CONFIG['epochs']
        )
        
        self.history = {'train_loss': [], 'val_loss': [], 'val_iou': []}
    
    def criterion(self, pred, target):
        return 0.5 * self.dice_loss(pred, target) + 0.5 * self.ce_loss(pred, target)
    
    def train_epoch(self, loader):
        self.model.train()
        total_loss = 0
        
        for images, masks in tqdm(loader, desc='Training'):
            images = images.to(self.device)
            masks = masks.to(self.device)
            
            self.optimizer.zero_grad()
            outputs = self.model(images)
            loss = self.criterion(outputs, masks)
            loss.backward()
            self.optimizer.step()
            
            total_loss += loss.item()
        
        return total_loss / len(loader)
    
    def validate(self, loader):
        self.model.eval()
        total_loss = 0
        total_iou = 0
        
        with torch.no_grad():
            for images, masks in loader:
                images = images.to(self.device)
                masks = masks.to(self.device)
                
                outputs = self.model(images)
                loss = self.criterion(outputs, masks)
                total_loss += loss.item()
                
                # Ù…Ø­Ø§Ø³Ø¨Ù‡ IoU
                preds = outputs.argmax(dim=1)
                iou = self.calculate_iou(preds, masks)
                total_iou += iou
        
        return total_loss / len(loader), total_iou / len(loader)
    
    def calculate_iou(self, pred, target):
        """Ù…Ø­Ø§Ø³Ø¨Ù‡ Mean IoU"""
        ious = []
        for cls in range(1, CONFIG['num_classes']):  # Skip background
            pred_cls = (pred == cls)
            target_cls = (target == cls)
            intersection = (pred_cls & target_cls).sum().float()
            union = (pred_cls | target_cls).sum().float()
            if union > 0:
                ious.append((intersection / union).item())
        return np.mean(ious) if ious else 0
    
    def train(self, train_loader, val_loader):
        """Ø¢Ù…ÙˆØ²Ø´ Ú©Ø§Ù…Ù„"""
        best_iou = 0
        
        for epoch in range(CONFIG['epochs']):

            val_loss, val_iou = self.validate(val_loader)

            # Ø°Ø®ÛŒØ±Ù‡ ØªØ§Ø±ÛŒØ®Ú†Ù‡
            self.history['train_loss'].append(train_loss)
