نمودار خطی و نقطه‌ای پیشرفته

نمودار خطی با چندین سری داده 

# داده چندین شهر
months = ['فروردین', 'اردیبهشت', 'خرداد', 'تیر', 'مرداد']
tehran = [25, 28, 32, 35, 38]
mashhad = [20, 23, 27, 30, 33]
shiraz = [22, 25, 29, 32, 35]

plt.figure(figsize=(12, 8))
plt.plot(months, tehran, 'o-', label='تهران', linewidth=2)
plt.plot(months, mashhad, 's-', label='مشهد', linewidth=2)
plt.plot(months, shiraz, '^-', label='شیراز', linewidth=2)

plt.xlabel('ماه')
plt.ylabel('دما (سانتی‌گراد)')
plt.title('مقایسه دمای شهرها')
plt.legend()
plt.grid(true, alpha=0.3)
plt.show()

نکات کاربردی

  • • استفاده از grid برای خوانایی بهتر
  • • همیشه legend را اضافه کنید
  • • از رنگ‌ها و مارکرهای متفاوت برای تمایز سری داده‌ها استفاده کنید
  • • اندازه نمودار را با figsize تنظیم کنید

نمودار میله‌ای و دایره‌ای

نمودار میله‌ای عمودی 

# داده امتیاز دانشجویان
students = ['علی', 'سارا', 'رضا', 'مریم', 'امیر']
scores = [85, 92, 78, 95, 88]

plt.figure(figsize=(10, 6))
bars = plt.bar(students, scores, color=['red', 'blue', 'green', 'orange', 'purple'])

# نمایش مقادیر روی میله‌ها
for bar in bars:
    height = bar.get_height()
    plt.text(bar.get_x() + bar.get_width()/2, height + 1, f'{height}', 
             ha='center', va='bottom')

plt.xlabel('دانشجو')
plt.ylabel('امتیاز')
plt.title('امتیاز دانشجویان')
plt.show()

ماتریس همبستگی 

import pandas as pd

# داده چندمتغیره
np.random.seed(42)
n = 100
data = {
    'x': np.random.normal(0, 1, n),
    'y': np.random.normal(0, 1, n),
    'z': np.random.normal(0, 1, n)
}
df = pd.dataframe(data)

# محاسبه ماتریس همبستگی
correlation = df.corr()

plt.figure(figsize=(8, 6))
plt.imshow(correlation, cmap='coolwarm', vmin=-1, vmax=1)
plt.colorbar(label='ضریب همبستگی')

plt.xticks(range(3), correlation.columns)
plt.yticks(range(3), correlation.columns)
plt.title('ماتریس همبستگی')
plt.show()

نمودارهای چندبخشی

ایجاد چند نمودار در یک صفحه 

import numpy as np

# داده مشترک
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)
y3 = np.exp(x/5)
y4 = np.log(x + 1)

# ایجاد نمودار 2x2
fig, axes = plt.subplots(2, 2, figsize=(12, 10))

# نمودار اول
axes[0, 0].plot(x, y1, 'r-')
axes[0, 0].set_title('sin(x)')

# نمودار دوم
axes[0, 1].plot(x, y2, 'b-')
axes[0, 1].set_title('cos(x)')

# نمودار سوم
axes[1, 0].plot(x, y3, 'g-')
axes[1, 0].set_title('exp(x/5)')

# نمودار چهارم
axes[1, 1].plot(x, y4, 'm-')
axes[1, 1].set_title('log(x+1)')

plt.tight_layout()
plt.show()

سفارشی‌سازی نمودارها

تنظیمات ظاهری 

import numpy as np

x = np.linspace(0, 10, 100)
y = np.sin(x)

plt.figure(figsize=(12, 8))

# تنظیمات استایل
plt.style.use('seaborn')

# رسم نمودار با تنظیمات پیشرفته
plt.plot(x, y, 
         color='#2e86ab', 
         linewidth=3, 
         linestyle='--', 
         marker='o', 
         markersize=8, 
         markerfacecolor='#a23b72', 
         markeredgecolor='#f18f01', 
         markeredgewidth=2)

# تنظیمات عنوان و برچسب‌ها
plt.title('نمودار سفارشی‌سازی شده', 
          fontsize=16, 
          fontweight='bold', 
          pad=20)

plt.xlabel('محور x', fontsize=12, fontweight='bold')
plt.ylabel('محور y', fontsize=12, fontweight='bold')

# تنظیمات محورها
plt.xlim(0, 10)
plt.ylim(-1.5, 1.5)

# تنظیمات گرید
plt.grid(true, alpha=0.3, linestyle=':')

# تنظیمات فونت
plt.rcparams['font.family'] = 'vazirmatn'
plt.rcparams['font.size'] = 12

plt.tight_layout()
plt.show()

ذخیره و export نمودارها

فرمت‌های مختلف خروجی 

import numpy as np

x = np.linspace(0, 10, 100)
y = np.sin(x)

plt.figure(figsize=(10, 6))
plt.plot(x, y)
plt.title('نمودار سینوس')

# ذخیره در فرمت‌های مختلف
plt.savefig('plot.png', dpi=300, bbox_inches='tight')
plt.savefig('plot.jpg', dpi=300, quality=95)
plt.savefig('plot.pdf', dpi=300, bbox_inches='tight')
plt.savefig('plot.svg', dpi=300, bbox_inches='tight')

# ذخیره با پس‌زمینه شفاف
plt.savefig('plot_transparent.png', 
            dpi=300, 
            bbox_inches='tight', 
            transparent=true)

plt.close()

تنظیمات پیشرفته ذخیره‌سازی

  • • dpi: کیفیت تصویر (نقطه در اینچ)
  • • bbox_inches: حاشیه‌های اضافی
  • • transparent: پس‌زمینه شفاف
  • • quality: کیفیت برای فرمت JPEG
  • • facecolor: رنگ پس‌زمینه
  • • edgecolor: رنگ حاشیه

# یک مثال جامع از matplotlib
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# ایجاد داده‌های مصنوعی
np.random.seed(42)
dates = pd.date_range(start='2023-01-01', end='2023-12-31', freq='d')
values = np.cumsum(np.random.randn(len(dates)) * 0.5 + 100)

# ایجاد نمودار زیبا
plt.figure(figsize=(15, 8))
plt.plot(dates, values, color='#2e86ab', linewidth=2)
plt.fill_between(dates, values, alpha=0.3, color='#2e86ab')

plt.title('روند داده‌ها در طول زمان', fontsize=16, fontweight='bold')
plt.xlabel('تاریخ', fontsize=12)
plt.ylabel('مقدار', fontsize=12)
plt.grid(true, alpha=0.3)
plt.tight_layout()
plt.show()
